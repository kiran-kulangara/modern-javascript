// Create a calculator // Create an object calculator with three methods: // read() prompts for two values and saves them as object properties with names a and b respectively. // sum() returns the sum of saved values. // mul() multiplies saved values and returns the result. let calculator = { a: null, b: null, read() { this.a = Number(prompt("Enter the first value?")); this.b = Number(prompt("Enter the second value?")); }, sum() { return (this.a + this.b); }, mul() { return (this.a * this.b) } }; calculator.read(); alert( calculator.sum() ); alert( calculator.mul() ); calculator.read(); console.log(calculator);
